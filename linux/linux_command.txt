//regex
*	匹配0或多个字符
?	匹配任意一个字符
[list]	匹配list中的任意单一字符
[!list]	匹配除list中的任意的那一字符之外的字符
[c1-c2]	匹配c1-c2中的任意单一字符 如：[0-9] [a-z]
{String1, String2, ..} 匹配string1或string2或更多其一字符串
{c2..c2}匹配c1-c2中全部字符 如：{1..10}

//linux用户管理
sudo adduser ljp	//增加新用户
sudo deluser ljp --remove-home	
			//删除用户
su ljp			//切换用户

groups ljp		//查看ljp用户属于哪个分组
cat /etc/group | sort //查看/etc/group 文件
cat /etc/group | grep -E "shiyanlou|sudo"  //搜索shiyanlou|sudo的项

sudo usermod -G sudo ljp
		//将ljp加入 sudo 用户组

//linux文件权限
ls -l 		//使用较长格式列出文件详细
ll		//使用较长格式列出文件详细
ls -A		//显示全部文件 包括隐藏文件
ls -dl <dirname>	//查看dirname的的完整属性
ls -AsSH		//s显示文件大小 S按文件大小排序

sudo  chown ljp iphone //变更文件iphone所有者为ljp

chmod 700 iphone	//修改文件权限为只允许所有者rwx  
		//顺序是（拥有者-所处用户组-其他用户）

chmod go-rw iphone //完成上述命令同样的功能
		//'g''o'还有'u'，分别表示group，
		//others，user，'+'，'-' 就分别表示增加和去掉相应的权限。
